INSTRUCTIONS: (within Console)

Create A Song & addToLibrary:

musicAPI.addToLibrary('title', 'artist', 'album', coverArtURL);

removeFromLibrary:

musicAPI.removeFromLibrary('title');

addToPlaylist:

musicAPI.addToPlaylist('title');

removeFromPlaylist:

musicAPI.removeToPlaylist('title');

shufflePlaylist

musicAPI.shufflePlaylist();

playSong

musicAPI.playSong('title');

pauseSong

musicAPI.pauseSong();

playnextSong

musicAPI.playnextSong();
